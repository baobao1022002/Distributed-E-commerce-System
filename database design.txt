Table users {
  id varchar [pk]
  customer_id varchar [unique]
  email varchar [unique]
  password varchar
  created_at timestamp
  updated_at timestamp
}

// Order Service DB
Table orders {
  id varchar [pk]
  customer_id varchar
  customer_email varchar
  status varchar
  total_amount decimal
  discount_amount decimal
  final_amount decimal
  coupon_code varchar
  correlation_id varchar
  payment_id varchar
  expires_at timestamp
  created_at timestamp
  is_deleted boolean
  created_by varchar
  created_date timestamp
  last_modified_by varchar
  last_modified_date last_modified_date
}

Table order_items {
  id varchar [pk]
  order_id varchar [ref: > orders.id]
  product_id varchar
  price int
  quantity int
  is_deleted boolean
  created_by varchar
  created_date timestamp
  last_modified_by varchar
  last_modified_date last_modified_date
}
